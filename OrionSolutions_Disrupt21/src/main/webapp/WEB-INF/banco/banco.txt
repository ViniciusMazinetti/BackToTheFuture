/** 
Criamos uma tabela de Personagens mais relevantes do filme "Back To The Future" chamada: "t_bttf_character", com os seguintes atributos:
"id_character", "nm_character", "ds_character" e "nm_actor".
 **/
CREATE TABLE t_bttf_character (
    id_character  NUMBER(3) NOT NULL,
    nm_character  VARCHAR2(60) NOT NULL,
    ds_character  VARCHAR2(1000) NOT NULL,
    nm_actor      VARCHAR2(60) NOT NULL
);

/** 
Colocamos na tabela "t_bttf_character", Uma chave Primária no atributo "id_character" para identificar o Personagem. 		
**/
ALTER TABLE t_bttf_character ADD CONSTRAINT pk_bttf_character PRIMARY KEY ( id_character );

/**
Criamos uma tabela associativa das tabelas: "t_bttf_character" e "t_bttf_object", cujo nome é:"t_bttf_character_object",
Pois vimos que era uma Relação N pra N, com o objetivo de identificar qual personagem usou aquele determinado objeto,
para que nós possamos buscar no banco de uma maneira mais fácil e assim, ajudar na Descrição do personagem no nosso site. Com os atributos:
"id_character(FK)" e "id_object(FK)". 
**/
CREATE TABLE t_bttf_character_object (
    id_character  NUMBER(3) NOT NULL,
    id_object     NUMBER(3) NOT NULL
);

/**
Colocamos na tabela "t_bttf_character_object", duas chaves Primárias Estrangeiras das tabelas: "t_bttf_character" e "t_bttf_object". 
**/
ALTER TABLE t_bttf_character_object ADD CONSTRAINT pk_bttf_character_object PRIMARY KEY ( id_character,
                                                                                          id_object );

/**
Criamos uma tabela associativa das tabelas: "t_bttf_character" e "t_bttf_location", cujo nome é: "t_bttf_event", pois vimos que seria uma relação N pra N, com o objetivo de identificar eventos que ocorre naquela localização com diferentes personagens. Com os atributos: "id_event(PK)" , "id_location(FK)", "id_character(FK)", "nm_event", "ds_event", "url_image" e "dt_event".  
**/
CREATE TABLE t_bttf_event (
    id_event      NUMBER(3) NOT NULL,
    id_location   NUMBER(4) NOT NULL,
    id_character  NUMBER(3) NOT NULL,
    nm_event      VARCHAR2(60) NOT NULL,
    ds_event      VARCHAR2(1000) NOT NULL,
    url_image     VARCHAR2(1000) NOT NULL,
    dt_event      DATE NOT NULL
);

/**
Colocamos na tabela "t_bttf_event", uma chave primária:"id_event", para identificar o evento e 2 chaves estrangeiras: "id_location" e "id_character".
**/
ALTER TABLE t_bttf_event ADD CONSTRAINT pk_bttf_event PRIMARY KEY ( id_event );

/**
Criamos uma tabela multivalorada chamada: "t_bttf_image_character", pois vimos que o personagem poderá ter 1 ou N imagens,
para assim armazenar várias imagens do mesmo personagem, com os atributos: "id_character(FK)", "id_image(PK)", "url_image" e "ds_image". 
**/
CREATE TABLE t_bttf_image_character (
    id_image      NUMBER(3) NOT NULL,
    id_character  NUMBER(3) NOT NULL,
    url_image     VARCHAR2(1000) NOT NULL,
    ds_image      VARCHAR2(1000) NOT NULL
);

/**
Colocamos na tabela "t_bttf_image_character", uma chave primária:"id_image", para identificar a imagem e uma chave Primária Estrangeira:"id_character", pois vimos
que essa tabela depende da tabela "t_bttf_character" para existir.
**/
ALTER TABLE t_bttf_image_character ADD CONSTRAINT pk_bttf_image_character PRIMARY KEY ( id_image,
                                                                                        id_character );
/**
Criamos uma tabela multivalorada chamada: "t_bttf_image_location", pois vimos que a localidade poderá ter 1 ou N imagens,
para assim armazenar várias imagens da mesma localidade, com os atributos: "id_image(PK)", "id_location(FK)", "url_image" e "ds_image". 
**/
CREATE TABLE t_bttf_image_location (
    id_image     NUMBER(3) NOT NULL,
    id_location  NUMBER(4) NOT NULL,
    url_image    VARCHAR2(1000) NOT NULL,
    ds_image     VARCHAR2(1000) NOT NULL
);

/**
Colocamos na tabela "t_bttf_image_location", uma chave primária:"id_image", para identificar a imagem e uma chave Primária Estrangeira:"id_location", pois vimos
que essa tabela depende da tabela "t_bttf_location" para existir.
**/
ALTER TABLE t_bttf_image_location ADD CONSTRAINT pk_bttf_image_location PRIMARY KEY ( id_image,
                                                                                      id_location );

/**
Criamos a Tabela de Localizações mais relevantes do filme "Back To The Future" chamada: "t_bttf_location", com os seguintes atributos: "id_location(PK)", "nm_location" e "ds_location".
**/
CREATE TABLE t_bttf_location (
    id_location  NUMBER(4) NOT NULL,
    nm_location  VARCHAR2(80) NOT NULL,
    ds_location  VARCHAR2(1000) NOT NULL

/**
Colocamos na tabela "t_bttf_location", uma chave primária no atributo: "id_location", para identificar a localização.  
**/
ALTER TABLE t_bttf_location ADD CONSTRAINT pk_bttf_location PRIMARY KEY ( id_location );

/**
Criamos a Tabela de Objetos mais importantes do filme "Back To The Future" chamada: "t_bttf_object",com os seguintes atributos: "id_object(PK)", "nm_object", "ds_object" e "url_image". 
**/
CREATE TABLE t_bttf_object (
    id_object  NUMBER(3) NOT NULL,
    nm_object  VARCHAR2(80) NOT NULL,
    ds_object  VARCHAR2(1000) NOT NULL,
    url_image  VARCHAR2(1000) NOT NULL
);

/**
Colocamos na tabela "t_bttf_location", uma chave primária no atributo: "id_object", para identificar o objeto.
**/
ALTER TABLE t_bttf_object ADD CONSTRAINT pk_bttf_object PRIMARY KEY ( id_object );
